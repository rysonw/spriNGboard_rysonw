cmake_minimum_required(VERSION 3.5 FATAL_ERROR)

## This is not recommended:
set(CMAKE_CXX_COMPILER "g++")

find_package(
  Catch2 3 REQUIRED
  PATHS ./catch_install
  NO_DEFAULT_PATH
)

    NAME test__regression
    COMMAND $<TARGET_FILE:test__regression>
